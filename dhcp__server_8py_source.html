265 def process_dhcp_message(self, giaddr = '00000000',message_type='00'):
266 if giaddr != '00000000':
267 #Server forwards all responses to DHCP relay messages to the BOOTP relay address
268 target_address = giaddr

270 #Server sends DHCPOFFER and DHCPACK messages to the individual client address self.client_ip_address
271 target_address = self.client_ip_address

273 #Server sends DHCPOFFER and DHCPACK messages to the broadcast address 0xffffffff
274 target_address = self.apply_find_broadcast(self.Configuration['IP_DHCP'], self.Configuration['MASK_DHCP'])
275
276 elif self.bootp_flags != '0000' and giaddr == '00000000' and self.client_ip_address == '00000000':

279 else:

293 def apply_find_broadcast(self, ip, mask):
294 #Convert the string representation of the IP address into a list of numbers
295 ip = [int(elem) for elem in ip.split('.')]
296
297 #Convert the string representation of the subnet mask into a list of numbers
298 mask = [int(elem) for elem in mask.split('.')]
299
300 #Perform bitwise NOT operation on each element of the subnet mask
301 inverse_mask = [~int(elem) & 0xFF for elem in mask]
302
303 #Apply the subnet mask to the IP address
304 result = [hex((ip[i]&mask[i])+inverse_mask[i])[2:] for i in range(0,4)]
305 arr_with_zero = ['0' + item if len(item) < 2 else item for item in result]
306 return ''.join(arr_with_zero)
