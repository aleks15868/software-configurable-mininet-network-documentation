100# @brief Создание экземпляра Mininet с контроллером.
101
102#Создается объект сети с указанным контроллером, который управляет сетью.
103 net = Mininet(controller=Controller)
104
105#Добавление контроллера в сеть.
106#Контроллер управляет работой свитча и взаимодействием между хостами.
107 net.addController('c0')
108
109# @brief Создание виртуального свитча.
110
111#Добавление свитча OVS (Open vSwitch) в сеть для соединения хостов.
112 switch = net.addSwitch('s1')
113
114# @brief Подключение физического интерфейса eth0 к виртуальному свитчу.
115
116#Физический интерфейс (eth0) подключается к виртуальному свитчу, чтобы обеспечить взаимодействие с внешней сетью.
117 Intf('eth0', node=switch)
118
119# @brief Добавление хостов в сеть.
120
121#Хосты представляют собой виртуальные машины с IP-адресами в указанном диапазоне.
122 h1 = net.addHost('h1', ip='192.168.2.3/24')
123 h2 = net.addHost('h2', ip='192.168.2.4/24')
124
125#Подключение хостов к свитчу через виртуальные линк-соединения.
126 net.addLink(h1, switch)
127 net.addLink(h2, switch)
128
129#Запуск виртуальной сети.
130 net.start()
131
132# @brief Настройка IP-адресов хостов вручную.
133
134#Использование команды ifconfig для установки IP-адресов и сетевых масок на виртуальных хостах.
135 h1.cmd('ifconfig h1-eth0 192.168.2.3 netmask 255.255.255.0')
136 h2.cmd('ifconfig h2-eth0 192.168.2.4 netmask 255.255.255.0')
